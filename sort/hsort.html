Heap Sort

Description:

Heap Sort uses a binary heap structure to repeatedly extract the maximum or minimum element and place it in its correct position.

#include<iostream>
using namespace std;
void heapify(int arr[], int n, int i)
{
    	int largest = i;
    	int left = 2 * i + 1;
    	int right = 2 * i + 2;
    	if (left < n && arr[left] > arr[largest])
    	    	largest = left;
    	if (right < n && arr[right] > arr[largest])
    	    	largest = right;
    	if (largest != i)
	{
    	    	swap(arr[i], arr[largest]);
    	    	heapify(arr, n, largest);
    	}
}

void hsort(int arr[], int n)
{
    	for (int i = n / 2 - 1; i >= 0; i--)
    	    	heapify(arr, n, i);
    	for (int i = n - 1; i > 0; i--)
	{
        	swap(arr[0], arr[i]);
        	heapify(arr, i, 0);
    	}
}

{
    	int n;
    	cout << "Enter size of array: ";

def heapify(arr, n, i):
    	largest = i
    	left = 2 * i + 1
    	right = 2 * i + 2
    	if left < n and arr[left] > arr[largest]:
    	    	largest = left
    	if right < n and arr[right] > arr[largest]:
    	    	largest = right
    	if largest != i:
    	    	arr[i], arr[largest] = arr[largest], arr[i]
    	    	heapify(arr, n, largest)

def hsort(arr):
    	n = len(arr)
    	for i in range(n // 2 - 1, -1, -1):
    	    	heapify(arr, n, i)
    	for i in range(n - 1, 0, -1):
    	    	arr[i], arr[0] = arr[0], arr[i]
    	    	heapify(arr, i, 0)

arr = eval(input("Enter list: "))
hsort(arr)
print("Sorted list:", arr)

Time Complexity:

  • Best Case: O[nlogn]
  • Worst Case: O[nlogn]
  • Average Case: O[nlogn]

Space Complexity:

  • O[1] (in-place, does not require additional space)

Used for:

  • Efficient, memory-conserving sorting.

Applications:

  • Priority Queues: Used in scheduling tasks.
  • Memory-Constrained Sorting: Requires minimal additional storage.
  • Real-Time Systems: Offers consistent performance characteristics.
