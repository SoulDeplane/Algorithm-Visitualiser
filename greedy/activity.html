Activity Selection

Description:

Given a set of activities with start and finish times, the goal is to select the maximum number of non-overlapping activities that can be scheduled within a time frame.

Code Implementation: 


#include<iostream>
#include<algorithm>
using namespace std;
struct Activity
{
    	int start, end;
};

bool compare(Activity a1, Activity a2)
{
    	return a1.end < a2.end;
}

void activitySelection(Activity arr[], int n)
{
    	sort(arr, arr + n, compare);
    	cout << "Selected Activities: \n";
    	int lastEnd = arr[0].end;
    	cout << "(" << arr[0].start << ", " << arr[0].end << ") ";
    	for (int i = 1; i < n; i++)
	{
    	    	if (arr[i].start >= lastEnd)
		{
    	    	    	cout << "(" << arr[i].start << ", " << arr[i].end << ") ";
    	    	    	lastEnd = arr[i].end;
    	    	}
    	}
    	cout << endl;
}

main()
{
    	int n;
    	cout << "Enter number of activities: ";
    	cin >> n;
    	Activity arr[n];
    	cout << "Enter start and end time of activities:\n";
    	for (int i = 0; i < n; i++) 
        	cin >> arr[i].start >> arr[i].end;
    	activitySelection(arr, n);
    	return 0;
}            


def activity_selection(activities):
    	activities.sort(key=lambda x: x[1])  
    	selected = [activities[0]]  
    	last_end = activities[0][1]
    	for i in range(1, len(activities)):
    	    	if activities[i][0] >= last_end:
    	    	    	selected.append(activities[i])
    	    	    	last_end = activities[i][1]
    	print("Selected Activities:", selected)

activities = eval(input("Enter list of (start, end) tuples: "))
activity_selection(activities)

Time Complexity:

  • O[nlogn] (due to sorting the activities by finish time)

Space Complexity:

  • O[1] (only a few extra variables are used)

Used for:

  • Efficient scheduling where maximum tasks need to be completed without overlapping.

Applications:

  • Event Scheduling: Ensuring the maximum number of events fit within a schedule.
  • Resource Allocation: Assigning tasks to workers where no two tasks overlap.
